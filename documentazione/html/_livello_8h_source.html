275  /*
276  * Ci dice se la cartaa a tessee a sinistra
277  * @param i, indice della tessera nel mazzo
278  */
279  bool haTessereSopra(int i){
280  unsigned r = mazzo[i].getRiga(),
281  c = mazzo[i].getColonna();
282  // se il livello di profondità della tessera nel mazzo alla posizione indice è 4
283  if(mazzo[i].getProfondita() == 4) return false;
284 
285  // se il livello di profondità della tessera nel mazzo alla posizione indice è 3
286  else if(mazzo[i].getProfondita() == 3)
287  return presente(r, c-1) || presente(r, c+1); // coperta dal livello 4
288 
289  // se il livello di profondità della tessera nel mazzo alla posizione indice è 2
290  else if(mazzo[i].getProfondita() == 2)
291  return presente(r-1,c) || presente(r+1, c) || presente(r+1, c+1) // coperta dal livello 4
292  ||
293  presente(r-1, c-1) || presente(r+1, c-1) || presente(r-1, c+1); // coperte dal livello 3
294 
295  // se il livello di profondità della tessera nel mazzo alla posizione indice è 1
296  return presente(r-1,c-1) || presente(r+1,c-1) || presente(r+1,c+1) // coperte dal livello 4
297  ||
298  presente(r-1,c) || presente(r+1,c) // coperte dal livello 3
299  ||
300  presente(r, c-1) || presente(r,c+1); // coperte dal livello 2
301  }
